**CTA handler (`handleBookNow`):**
```typescript
const handleBookNow = () => {
  // Validate required fields
  const newErrors: Record<string, string> = {}
  if (!origin) newErrors.origin = 'required'
  if (tripType !== 'hourly' && !destination) newErrors.destination = 'required'
  if (!date) newErrors.date = 'required'
  if (!time) newErrors.time = 'required'
  if (Object.keys(newErrors).length > 0) {
    setErrors(newErrors)
    return
  }
  setErrors({})

  // Write to Zustand store
  const store = useBookingStore.getState()
  store.setTripType(tripType)
  store.setOrigin(origin)
  if (tripType === 'hourly') {
    // hours already set by DurationSelector
  } else {
    store.setDestination(destination)
  }
  store.setPickupDate(date)
  store.setPickupTime(time)

  // Deep-link to Step 3
  useBookingStore.setState({
    currentStep: 3,
    completedSteps: new Set([1, 2]),
  })

  router.push('/book')
}
```

Note: `tripType` and `hours` are read from store (set by TripTypeTabs and DurationSelector respectively). `origin`, `destination`, `date`, `time` are local state written to store only on CTA click.

Field gap between form rows: 24px (same as Step1TripType pattern). Label elements use `.label` class on `<label>` elements.
