async function getAdminUser() {
  const supabase = await createClient()

  if (error || !user) return { user: null, error: '401' as const }

  return { user, error: null }
}

const geojsonFeatureSchema = z.object({
  type: z.literal('Feature'),
  geometry: z.object({
    type: z.literal('Polygon'),
    coordinates: z.array(z.array(z.array(z.number()).length(2))).min(1),
  }),
  properties: z.record(z.unknown()).optional().nullable(),
})

const zoneCreateSchema = z.object({
  name: z.string().min(1).max(100),
  geojson: geojsonFeatureSchema,
})

const zoneToggleSchema = z.object({
  id: z.string().uuid(),
  active: z.boolean(),
})

**GET handler:**
1. Auth guard — return 401/403 on error
2. Parse query params from `new URL(request.url).searchParams`:
   - `page`: `Math.max(0, parseInt(searchParams.get('page') ?? '0', 10))` — fallback 0, clamp to >= 0
   - `limit`: `Math.min(100, Math.max(1, parseInt(searchParams.get('limit') ?? '20', 10)))` — fallback 20, clamp 1-100
   - `startDate`: `searchParams.get('startDate')` — ISO date string or null
   - `endDate`: `searchParams.get('endDate')` — ISO date string or null
   - `tripType`: `searchParams.get('tripType')` — one of 'one-way', 'airport-pickup', etc. or null
   - `search`: `searchParams.get('search')` — free text or null
3. Build query:
   ```typescript
   const supabase = createSupabaseServiceClient()
   let query = supabase
     .from('bookings')
     .select('*', { count: 'exact' })
     .order('created_at', { ascending: false })
     .range(page * limit, page * limit + limit - 1)
   ```
4. Apply conditional filters:
   - `if (startDate) query = query.gte('pickup_date', startDate)`
   - `if (endDate) query = query.lte('pickup_date', endDate)`
   - `if (tripType) query = query.eq('trip_type', tripType)`
   - `if (search) query = query.or(\`client_first_name.ilike.%${search}%,client_last_name.ilike.%${search}%,booking_reference.ilike.%${search}%\`)`
5. Execute query, check for error — return 500 if error
6. Return `NextResponse.json({ bookings: data, total: count ?? 0, page, limit })`

Note on `pickup_date` filtering: the column is `text NOT NULL` storing `'YYYY-MM-DD'` strings. Lexicographic `.gte`/`.lte` comparison works correctly for ISO date format.
